interface MonteCarloSummary {
  baseline_p10: number;
  baseline_p50: number;
  baseline_p90: number;
  graph_aware_p10: number;
  graph_aware_p50: number;
  graph_aware_p90: number;
  baseline_evar_95: number | null;
  graph_aware_evar_95: number | null;
}

interface TradeoffEntry {
  bom_name: string;
  losing_axis: string;       // "cost" | "eta" | "risk"
  baseline_value: number;
  graph_aware_value: number;
  delta_pct: number;
  narrative: string;
}

interface BomDelta {
  bom_name: string;
  cost_delta_pct: number;
  eta_delta_pct: number;
  co2_delta_pct: number;
  cascade_risk_delta_pct: number;
}

interface BenchmarkSummary {
  run_id: number;
  run_tag: string;
  timestamp: string;
  n_boms: number;
  cost_delta_pct: number;
  eta_delta_pct: number;
  co2_delta_pct: number;
  cascade_risk_delta_pct: number;
  monte_carlo: MonteCarloSummary;
  tradeoff: TradeoffEntry;
  bom_deltas: BomDelta[];
  feeds_fallback: boolean;
  noise_floor_pct: number;
}

interface FiedlerPoint {
  step: number;
  removed: number | null;
  removed_name: string | null;
  lambda2: number;
  delta_pct: number;
  collapsed_boms: string[];
}

interface FiedlerCurveData {
  points: FiedlerPoint[];
  baseline_lambda2: number;
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → API | benchmarkAPI calls cross the browser/API boundary; JWT auth header sent by axios interceptor |
| API response → DOM | JSON delta values and narrative strings rendered into DOM |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-03-01 | Information Disclosure | benchmarkAPI.summary() in frontend | accept | No user PII in benchmark data — all aggregate analytics. JWT token is sent by axios interceptor but endpoint is public (accept per 04-02 T-04-02-03). No secrets exposed to frontend bundle. |
| T-04-03-02 | Tampering | `summary.tradeoff.narrative` string from API rendered as text | mitigate | Render as `{summary.tradeoff.narrative}` JSX text node — NOT `dangerouslySetInnerHTML`. String is server-generated, not user-controlled, but defense-in-depth: treat as untrusted text. |
| T-04-03-03 | Denial of Service | Fiedler drawer renders large `collapsed_boms` list | accept | `collapsed_boms` is bounded by n_boms (10 items max). No virtualization needed. Accept. |
| T-04-03-04 | Information Disclosure | `feeds_fallback` flag leaks server config state | accept | Only reveals whether external feeds were available at run time — no API keys, no internal URLs. Accept per public-dashboard principle. |
</threat_model>
